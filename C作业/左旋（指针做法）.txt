void Exchange(char *str, int k)
{
	int i,m;
	char temp;
	char *sta;
	for (i = 0; i < k; i++)
	{
		sta = str;
		temp = *str;
		for (m = 0; m <= (int)strlen(str)-1; m++)
		{
			*sta = *(sta + 1);
			sta++;
		}
		*(str+strlen(str)) = temp;
	}
}

int main()
{
	int leng;
	printf("请输入要左旋几个字符\n");
	scanf("%d", &leng);
	char str[] = "ABCDEFG";
	Exchange(str,leng);
	printf("%s\n", str);
	system("pause");
	return 0;
}
