#include<stdio.h>
#include<stdlib.h>
void Exchange(char str[],int leng,int length)
{
	char temp;
	int i,m;
	for (i = 1; i <= leng; i++)
	{
		temp = str[0];
		for (m = 0; m < length-1; m++)
		{
			str[m] = str[m+1];
		}
		str[length - 1] = temp;
	}
}
int main()
{
	int leng;
	int length;
	printf("请输入要左旋几个字符\n");
	scanf("%d", &leng);
	char str[] = "ABCDEFG";
	length = sizeof(str)/sizeof(str[0])-1;
	Exchange(str,leng,length);
	printf("%s\n", str);
	system("pause");
	return 0;
}
